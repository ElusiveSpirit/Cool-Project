Основные команды для гита:

[текст] - голый параметр (без квадратных скобок)

git status
git add [файл]                      // Добавить файлы для коммита
git reset --HEAD [файл]             // Перестать отслеживать файлы (После команды add)
git checkout [ветка или коммит]     // Перейти на существующую ветку или коммит
git checkout -b [название]          // Создать новую ветку и перейти на неё
git branch                          // Показывает все ветки
git merge --no-ff [ветка или коммит]// Мёрджит выбранную ветку на текущую
git push origin [ветка]             // Слить ветку на гитхаб
git pull                            // Скачать с гитхаба новые коммиты
git branch -d [ветка]               // Удаление ветки

git commit -m "[текст]"             // Коммит с сообщением
git commit --amend                  // Добавить файлы в последний коммит или изменить сообщение
